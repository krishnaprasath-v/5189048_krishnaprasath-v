#include <stdio.h>

int main() {
    int n, i, j;
    scanf("%d", &n);
    int s[n];
    for(i = 0; i < n; i++) {
        scanf("%d", &s[i]);
    }
    int d, m;
    scanf("%d %d", &d, &m);
    int count = 0;
    for(i = 0; i <= n - m; i++) {
        int sum = 0;
        for(j = i; j < i + m; j++) {
            sum = sum + s[j];
        }
        if(sum == d) {
            count++;
        }
    }
    printf("%d", count);
    return 0;
}
