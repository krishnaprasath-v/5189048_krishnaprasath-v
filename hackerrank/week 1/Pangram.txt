#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char s[1000];
    int letters[26] = {0};
    int i, index;
    
    fgets(s, sizeof(s), stdin);
    
    for(i = 0; s[i] != '\0'; i++) {
        if(isalpha(s[i])) {
            index = tolower(s[i]) - 'a';
            letters[index] = 1;
        }
    }
    
    for(i = 0; i < 26; i++) {
        if(letters[i] == 0) {
            printf("not pangram");
            return 0;
        }
    }
    
    printf("pangram");
    return 0;
}
