Grid challange
#include <stdio.h>
#include <string.h>

int main() {
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        char grid[n][105];
        for(int i=0;i<n;i++){
            scanf("%s",grid[i]);
        }

        for(int i=0;i<n;i++){
            int len=strlen(grid[i]);
            for(int j=0;j<len-1;j++){
                for(int k=j+1;k<len;k++){
                    if(grid[i][j]>grid[i][k]){
                        char temp=grid[i][j];
                        grid[i][j]=grid[i][k];
                        grid[i][k]=temp;
                    }
                }
            }
        }

        int ok=1;
        for(int c=0;c<n;c++){
            for(int r=0;r<n-1;r++){
                if(grid[r][c]>grid[r+1][c]){
                    ok=0;
                    break;
                }
            }
            if(ok==0) break;
        }

        if(ok) printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
